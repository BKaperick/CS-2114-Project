

public void Save(String filename, String data)
{
	FileOutputStream outputStream;

	try 
	{
		outputStream = openFileOutput(filename, Context.MODE_PRIVATE);
		outputStream.write(data.getBytes());
		outputStream.close();
	}	
	catch(Exception e)	
	{
		e.printStackTrace();
	}
}
